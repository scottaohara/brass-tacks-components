Additional Silent Display Classes

// Hide text that you don't want visible
// like on buttons that use an icon instead of text
%txt-hide {
  color: transparent;
  font: 0/0 a;
  text-shadow: none;
}

// Hides from screen readers and browsers
%hidden {
  display: none !important;
  height: 0;
  visibility: hidden;
}
